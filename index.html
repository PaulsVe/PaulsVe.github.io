SQL Schema & Queries

CREATE TABLE streaming_history (
  id INTEGER PRIMARY KEY,
  track_name TEXT,
  artist_name TEXT,
  end_time DATETIME,
  ms_played INTEGER
);

Sample Query:

SELECT artist_name, SUM(ms_played)/60000 AS total_minutes FROM streaming_history GROUP BY artist_name ORDER BY total_minutes DESC LIMIT 10;
